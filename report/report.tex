% -------------------------------------------------------
%  ЗАГОЛОВКИ РАЗДЕЛОВ
%
%  \section{Заголовок}  — нумерованный раздел: «N  Заголовок», по центру;
%    автоматически попадает в оглавление с номером.
%  \section*{Заголовок} — ненумерованный раздел, по центру;
%    в оглавление НЕ попадает — используй \addtoc{} вручную.
%  \subsection{...}     — нумерация вида «N.M», выравнивание влево.
%
%  \addtoc{Заголовок}   — вспомогательная команда: ставит \phantomsection
%    и добавляет запись уровня section в оглавление.
%    Вызывать непосредственно перед соответствующим \section*{}.
% -------------------------------------------------------
\usepackage{titlesec}

\setcounter{secnumdepth}{2}

\titleformat{\section}[hang]
    {\normalfont\filcenter}{\thesection}{0.7em}{}
\titleformat{name=\section, numberless}[hang]
    {\normalfont\filcenter}{}{0pt}{}
\titlespacing*{\section}{0pt}{18pt}{14pt}

\titleformat{\subsection}[hang]
    {\normalfont}{\thesubsection}{0.7em}{}
\titlespacing*{\subsection}{0pt}{14pt}{6pt}

\newcommand{\addtoc}[1]{%
    \phantomsection
    \addcontentsline{toc}{section}{#1}%
}
